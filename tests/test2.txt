programa foreveralone; 
var
	char op;
	float v1, v2;
	int i;

funcion float calc ()
{
	si (op == '+') entonces
	{
		regresa(v1 + v2);
	}
	si (op == '-') entonces
	{
		regresa(v1 - v2);
	}
	si (op == '*') entonces
	{
		regresa(v1 * v2);
	}
	si (op == '/') entonces
	{
		regresa(v1 / v2);
	}
}

principal()
{
	escribe("Escribe el operando (+, -, *, /):");
	lee(op);

	escribe("Escribe el valor 1: ");
	lee(v1);
	escribe("Escribe el valor 2: ");
	lee(v2);

	escribe("Resultado:", calc());

	desde i = 4 hasta 9 hacer
	{
		escribe("hola");
	}
}
